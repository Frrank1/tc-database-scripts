select pr.final_score from project_result pr, project p
where pr.project_id = p.project_id and pr.final_points > 0 
and pr.valid_submission_ind = 1 and
    pr.rating_ind = 1 and
    p.digital_run_ind = 1 and 
    p.status_id = 7 and
    p.phase_id = @ph@ and
    p.stage_id = @staid@ and 
    pr.user_id = @cr@
order by pr.final_score desc
