SELECT u.email AS email_address
     , u.handle
  FROM email_user u
     , user_notify_xref n
     , coder c
 WHERE u.status = 'A'
   AND u.user_id = n.user_id
   AND u.user_id = c.coder_id
   AND n.notify_id = 1
   AND c.coder_type_id = 1
   AND u.user_id NOT IN
(select user_id from user_terms_of_use_xref where terms_of_use_id = 20504)
UNION
SELECT u.email AS email_address
     , u.handle
  FROM email_user u
     , user_notify_xref n
     , coder c
 WHERE u.status = 'A'
   AND u.user_id = n.user_id
   AND u.user_id = c.coder_id
   AND n.notify_id = 1
   AND c.coder_type_id = 1
   AND u.user_id IN
(select coder_id from room_result where round_id IN (10093,10094) and advanced='Y' UNION
select coder_id from invite_list where round_id = 10102)
