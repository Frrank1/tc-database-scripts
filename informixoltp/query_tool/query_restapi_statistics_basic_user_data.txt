SELECT u.handle, c.country_name, coder.member_since, coder.quote,
 (CASE WHEN r.rating<0 THEN 't' ELSE 'f' END) AS is_pm,
 (CASE WHEN cp.user_id>0 THEN 't' ELSE 'f' END) AS is_copilot,
 NVL(cp.is_software_copilot, 'f') AS is_software_copilot,
 NVL(cp.is_studio_copilot, 'f') AS is_studio_copilot,
 i.link AS photo_link
FROM user AS u
JOIN coder ON u.user_id = coder.coder_id
JOIN country AS c ON c.country_code = coder.comp_country_code
JOIN rating AS r ON r.coder_id = coder.coder_id
LEFT JOIN tcs_catalog:copilot_profile AS cp ON u.user_id = cp.user_id
JOIN coder_image_xref AS cix ON cix.coder_id = u.user_id 
JOIN image AS i ON i.image_id = cix.image_id
WHERE u.handle = '@ha@'
