select u.handle, u.email as email_address
  from email_user u
     , room_result rr
 where rr.advanced = 'Y'
   and rr.coder_id not in (select user_id from group_user where group_id in (13,14))
   and rr.round_id = 12012
   and rr.coder_id = u.user_id
   and u.status = 'A'
  and u.user_id not in (select u.user_id
  from email_user u
     , room_result rr
 where rr.advanced = 'Y'
   and rr.coder_id not in (select user_id from group_user where group_id in (13,14))
   and rr.round_id = 12008
   and rr.coder_id = u.user_id
   and u.status = 'A')
