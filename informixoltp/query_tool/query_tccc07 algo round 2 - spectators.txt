SELECT u.email AS email_address
     , u.handle
  FROM email_user u
     , user_notify_xref n
 WHERE u.user_id NOT IN (
select u.user_id
  from email_user u
     , room_result rr
 where rr.advanced = 'Y'
   and rr.coder_id not in (select user_id from group_user where group_id in (13,14))
   and rr.round_id IN (10897,10898,10899,10906)
   and rr.coder_id = u.user_id
   and u.status = 'A')
   AND u.user_id = n.user_id
   AND u.status = 'A'
   AND n.notify_id = 1
