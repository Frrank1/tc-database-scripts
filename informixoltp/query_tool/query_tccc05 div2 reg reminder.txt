SELECT u.email AS email_address
     , u.handle
  FROM email_user u
     , coder_notify n
     , coder c
     , rating r
 WHERE u.user_id = n.coder_id
   and c.coder_id = u.user_id
   and u.user_id= r.coder_id
   and r.rating < 1200
   and c.coder_type_id = 1
   AND u.status = 'A'
   AND n.notify_id = 1
   and u.user_id not in (
select user_id
  from user_terms_of_use_xref 
 where terms_of_use_id in (22,23))
