SELECT u.email AS email_address
     , u.handle
  FROM email_user u
     , user_notify_xref n
 WHERE u.user_id NOT IN (select coder_id as user_id from invite_list where round_id = 10897
   UNION select user_id from event_registration where event_id = 3000)
   AND u.user_id = n.user_id
   AND u.status = 'A'
   AND n.notify_id = 1
