SELECT u.email AS email_address
     , u.handle
  FROM email_user u
     , coder_notify n
 WHERE u.user_id = n.coder_id
   AND u.status = 'A'
   AND n.notify_id = 1
   and u.user_id not in (
select u.user_id
  from user u
     , room_result rr
 where u.user_id = rr.coder_id
   and rr.round_id = 5879
   and rr.advanced = 'Y'
   and u.status = 'A')
