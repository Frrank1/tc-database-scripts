SELECT u.email AS email_address
     , u.handle
  FROM email_user u
     , user_notify_xref n
 WHERE u.user_id = n.user_id
   AND u.status = 'A'
   AND n.notify_id = 1
   AND u.handle NOT IN    
(SELECT u.handle
FROM email_user u
WHERE u.user_id IN
(select user_id from event_registration where event_id = 3001))
UNION
select u.handle, u.email as email_address
  from email_user u
     , room_result rr
 where rr.advanced = 'Y'
   and rr.coder_id not in (select user_id from group_user where group_id in (13,14))
   and rr.round_id IN (10888,10881)
   and rr.coder_id = u.user_id
   and u.status = 'A'
