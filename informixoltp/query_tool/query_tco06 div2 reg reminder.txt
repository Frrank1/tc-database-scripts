SELECT u.email AS email_address
     , u.handle
  FROM email_user u
     , coder_notify n
     , rating r
 WHERE u.user_id = n.coder_id
   AND u.status = 'A'
   AND n.notify_id = 1
   and u.user_id = r.coder_id
   and r.rating < 1200
   and not exists (select '1' 
                     from user_terms_of_use_xref x 
                    where user_id = u.user_id 
                      and x.terms_of_use_id in (20383))
