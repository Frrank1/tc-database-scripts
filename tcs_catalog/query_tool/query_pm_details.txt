select *
  from resource rur
     , user u
     , email e
     , resource_info ri
 where rur.project_id = @pj@
   and rur.resource_role_id in (13, 14)
   and rur.resource_id = ri.resource_id
   and ri.resource_info_type_id = 1
   and ri.value = u.user_id
   and e.user_id = u.user_id
   and e.primary_ind = 1

   -- exclude "Components", "Applications" and "LCSUPPORT" dummy users
   and u.user_id not in (22719217, 22770213, 22873364)
