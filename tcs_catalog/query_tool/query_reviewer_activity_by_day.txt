select to_char(ra.create_date, '%Y.%m.%d')
     , sum(case when cc.root_category_id = 9926572 then 0 else 1 end) as components
     , sum(case when cc.root_category_id = 9926572 then 1 else 0 end) as applications
  from rboard_application ra, project p, comp_versions cv, comp_catalog cc
 where cc.component_id = cv.component_id
   and cv.comp_vers_id = p.comp_vers_id
   and p.cur_version = 1
   and p.project_id = ra.project_id
 group by 1
 order by 1 desc
