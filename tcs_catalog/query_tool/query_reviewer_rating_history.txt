SELECT
p.project_id,
pi6.value as project_name,
(CASE WHEN rfd.score = 0 THEN "Bad" WHEN rfd.score = 1 THEN "Average" WHEN rfd.score = 2 THEN "Good" ELSE "Unknown" END) as score,
rr.rating as new_rating,
rr.review_date
FROM reviewer_rating rr
INNER JOIN project p ON p.project_id=rr.project_id
INNER JOIN project_info pi6 ON pi6.project_id=p.project_id and pi6.project_info_type_id=6
INNER JOIN review_feedback rf ON rf.project_id=p.project_id
INNER JOIN review_feedback_detail rfd ON rfd.review_feedback_id=rf.review_feedback_id and rfd.reviewer_user_id=rr.user_id
WHERE
p.project_category_id = @pt@ and
rr.user_id = @uid@
order by @sc@ @sd@, rr.review_date desc
