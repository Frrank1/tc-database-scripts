 SELECT
    s.submission_id,
    s.upload_id,
    s.screening_score,
    s.initial_score,
    s.final_score,
    s.submission_status_id,
    s.placement,
    s.create_date,
   replace(replace(replace(replace(replace(multiset (SELECT ric.content FROM review rv, resource r, review_item ri, review_item_comment ric
      WHERE rv.submission_id = s.submission_id AND rv.resource_id = r.resource_id AND r.resource_role_id = 20
        AND rv.review_id = ri.review_id AND ri.review_item_id = ric.review_item_id group by ric.content having count(ric.content) = 1)::lvarchar, 
'MULTISET{ROW(''', ''), 'ROW(''', ''), ''')}', ''), ''')', ''), 'MULTISET{}', '') as feedback
FROM
    project_prize_xref ppx, prize p, submission s
    INNER JOIN upload u ON s.upload_id = u.upload_id
WHERE u.project_id = @pj@
AND   u.upload_type_id = 1
AND   s.submission_type_id = 3
AND   s.submission_status_id in (1,4)
AND   p.prize_id = ppx.prize_id
AND   ppx.project_id = @pj@
AND   p.prize_type_id = 14
ORDER BY NVL(s.placement, 1000), s.create_date
